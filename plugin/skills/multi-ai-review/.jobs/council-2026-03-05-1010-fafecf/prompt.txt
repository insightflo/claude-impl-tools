## 파일 내용

### workflow-guide/SKILL.md (핵심 알고리즘 부분)
ALGORITHM get_recommendation():

  # ① 복구 체크 (최우선)
  IF .claude/orchestrate-state.json EXISTS:
    RETURN "/recover"  # 이후 /orchestrate-standalone --resume 안내

  IF git dirty OR git conflicts:
    RETURN "/recover"

  # ② 태스크 파일 체크
  IF TASKS.md NOT EXISTS:
    IF docs/planning/06-tasks.md EXISTS:
      RETURN "/tasks-migrate"
    ELSE:
      RETURN "/tasks-init"

  # ③ 유지보수 체크 ← 거버넌스 체크보다 반드시 먼저
  #    source_code = package.json/pyproject.toml/requirements.txt/src/ 중 하나 이상 존재
  IF source_code EXISTS:
    IF incomplete_tasks > 0:
      RETURN "/agile iterate"   # 기존 코드 수정/확장 → iterate 사용
    IF all_tasks_completed:
      RETURN "/audit"

  # ④ 거버넌스 체크 ← source_code 없는 신규 프로젝트 전용
  #    DOMAIN_COUNT는 TASKS.md 'domain:' 필드 기반; 없으면 TASK_COUNT>=30으로 대체
  IF TASK_COUNT >= 10 AND (DOMAIN_COUNT >= 2 OR TASK_COUNT >= 30) AND GOVERNANCE_DONE == "no":
    RETURN "/governance-setup"
    # ⚠️ AGENT_COUNT, TASKS.md 목적·설명 필드, 사용자 맥락은 이 조건에 영향 없음

  # ⑤ 인프라 체크 (거버넌스 완료 후)
  IF GOVERNANCE_DONE == "yes" AND TASK_COUNT >= 30 AND AGENT_COUNT == 0:
    RETURN "project-team/install.sh --mode standard"
    # 참고: claude-imple-skills 클론 디렉토리 내 project-team/ 에서 실행

  # ⑥ 신규 구현 (거버넌스 완료 + 인프라 준비, 코드 없음)
  IF GOVERNANCE_DONE == "yes" AND AGENT_COUNT > 0:
    IF TASK_COUNT >= 80: RETURN "/orchestrate-standalone --mode=wave"
    IF TASK_COUNT >= 30: RETURN "/orchestrate-standalone"
    ELSE: RETURN "/agile auto"

  # ⑦ 소규모 신규 구현 (거버넌스 불필요, 코드 없음)
  IF TASK_COUNT > 0 AND TASK_COUNT < 30:
    RETURN "/agile auto"

  # ⑧ 완료
  IF all_tasks_completed:
    RETURN "/audit"
```
