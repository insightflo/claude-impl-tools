### 수정 파일 1: orchestrate.sh (lines 175-232) — Sprint 블록

```bash
if [ "$MODE" = "sprint" ]; then
    header "Executing Sprint Mode (Workers: $WORKER_COUNT, Sprint Size: $SPRINT_SIZE)"

    # Bug #8: resume이 아닐 때만 planner 실행
    if [ "$RESUME" != "true" ]; then
        if ! "$NODE_CMD" "$SCRIPT_DIR/sprint-planner.js" "$TASKS_FILE" "$SPRINT_SIZE"; then
            log_error "Sprint planning failed or cancelled"
            exit 1
        fi
    fi

    # Bug #7: Sprint 루프
    while true; do
        SPRINT_STATE_VAL=$(node -e "
            try {
                const s = require('$PROJECT_DIR/.claude/sprint-state.json');
                console.log(s.state);
            } catch(e) { console.log('UNKNOWN'); }
        " 2>/dev/null || echo 'UNKNOWN')

        if [ "$SPRINT_STATE_VAL" = "UNKNOWN" ]; then
            log_error "Cannot read sprint-state.json (missing or corrupted)"
            exit 1
        fi

        # Bug #10: terminal states만 loop 종료
        if [ "$SPRINT_STATE_VAL" = "PI_COMPLETE" ] || [ "$SPRINT_STATE_VAL" = "PAUSED" ]; then
            log_info "Sprint state: $SPRINT_STATE_VAL — exiting loop"
            break
        fi

        # Bug #9: set -e 환경에서 exit code 캡처
        "$NODE_CMD" "$SCRIPT_DIR/sprint-runner.js" "$TASKS_FILE" && SPRINT_EXIT=0 || SPRINT_EXIT=$?

        # Bug #2: exit code별 분기
        case $SPRINT_EXIT in
            0) log_success "Sprint completed" ;;
            2)
                log_warn "Modification requested."
                log_info "Re-run: $0 --mode=sprint --resume"
                exit 2
                ;;
            3)
                log_info "Sprint stopped by user."
                exit 0
                ;;
            *)
                log_error "Sprint runner failed (exit code: $SPRINT_EXIT)"
                exit 1
                ;;
        esac
    done

    header "Orchestration Complete"
    log_success "Sprint mode execution finished!"
    exit 0
fi
```
